--To access the data from hstore column we use operator (->)

--if we want to access hstore values in filter condition

--if i want to add new key value pair admission =>Yes/no

update clg_info SET clg_info = clg_info || '"admission" => "Yes"' where cid =1
update clg_info SET clg_info = clg_info || '"admission" => "NO"' where cid =2

--key word inside update statement

delete from clg_info where clg_info ='"admission"=>"Yes"'

update clg_info set clg_info = delete (clg_info,'admission')
where clg_info -> 'City' ='Warangal'

select * from clg_info

--3.UUID
--UUID stands for Universal unique identifier
--UUID is 128 -bit quantity and it is generated by algorith.
--UUID is a sequence of 32 digit of hexadecimalvalue.

--by using two algoriths we can generate UUID values in column
--1.uuid_generate_v1()
--2.uuid_generate_v4()

--to implement or use above algorithms we need to create uuid extension then only it works
create extension "uuid-ossp"
--ossp - open source software project

--1.uuid_generate_v1()
--By using this algorithm we can generate UUID with combination of MAC address,
--Current Time stamp and random value.

--"58ff2ade-4ef2-11ed-80ab-7c70db33ba11",618cb73e-4ef2-11ed-80ac-7c70db33ba11

--4.JSON and JSONB
--JSON - Java Script object notation 
--JSON is open standared format that consists of key-value pairs.

JSON
The json data type works better than jsonb when:
--You have to perform a lot of INSERT operations
--You do not expect to perform complex queries on your JSON data
--You need to preserve the original JSON data indentation and format

--for ex: key => value by using hstore data type but we cant store key=> key => value.
--but in JSON we can store key:value ,key :{key:value,key:value}

JSONB
--jsonb is based on an optimized format that enables many new operations
--Set a new key
--Update the value of an existing key
--Set a value in a nested object
--Update the value of a nested key
--Delete a key
--Delete a nested key
--Concatenate JSON objects
--Deal with JSON array


create table orders(
id serial primary key,
Order_info json not Null)

select * from orders

--single row 
insert into orders (Order_info) values 
('{"Customer":"Mohit","items":{"Product":"Computer","qty":4}}')

insert into orders (Order_info) values 
('{"Customer":"Sohan","items":{"Product":"Mobile","qty":6}}'), 
('{"Customer":"Amit","items":{"Product":"Digital Watch","qty":8}}')


--How to extract JSON data type column data?
--to extrct the values from JSON data type column we use below opeartors
--: -> = It will return object field of JSON Key
--: ->> = It will return JSON Object filed by Text.

select * , order_info -> 'Customer' as C_Name from orders --JSON

select * , order_info ->> 'Customer' as C_Name from orders --Text

select * , order_info -> 'items' ->> 'Product' as P_Name from orders

--we can use aggrgate function on JSON data type column.
--min,max,count,avg,sum etc 
select min(order_info->'items'->>'qty') as min_Value from orders
