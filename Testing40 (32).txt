--It is just multiplication type of operation.

--AND opration
--A			B			O/P
--Match		Match		Display Record
--Unmatch	Match		Display Blank
--Match		Unmatch		Display Blank
--Unmatch	Unmatch		Display Blank

--2.OR
--It compares two columns in table row wise but here if both the input matches and either of the input match
--in a row of table then it will display that record from table.
--It will act as addition operation.

--OR Opeartion
--A			B			O/P
--Match		Match		Display A and B Record
--Unmatch	Match		Display B Record
--Match		Unmatch		Display A Record
--Unmatch	Unmatch		Display Blank Record

select * from STUDENT where FIRSTNAME = 'Sumit' OR lastname = 'Bhosle'--Display A Record

select * from STUDENT where FIRSTNAME = 'Sumit' OR lastname = 'More' --Display A and B Record

--Function() over (Partition by col1,col2...coln)

--Functions like count(),AVG(),MAX(),MIN(),SUM(),RANK(),DENSE_RANK() and ROW_NUMBER etc  

create table OVER_TEST(OID int, FirstName Varchar(20),Gender varchar(2),salary int)

--calculating Aggregated data based on gender column
select Gender, COUNT(*) as TotalCount,AVG(salary) as AvgSal,MAX(salary) as MaxSal,MIN(salary) as MinSal
,sum(salary) as SumSal from OVER_TEST
group by Gender

--Non-Aggregated data 
select *
from OVER_TEST A1
inner join 
(select gender,COUNT(*) as TotalCount,AVG(salary) as AvgSal,MAX(salary) as MaxSal,MIN(salary) as MinSal
,sum(salary) as SumSal from OVER_TEST
group by Gender) as B1
On A1.Gender = B1.Gender

--using over clause and partition by clause.
--Function() over (Partition by col1,col2...coln)
select OID,FirstName,Gender,salary
,count(gender) over (partition by gender) as TotalCount
,AVG(salary) over (Partition by gender) as Avgsal
,MAX(salary) over (Partition by gender) as Maxsal
,MIN(salary) over (Partition by gender) as Minsal
,SUM(salary) over (Partition by gender) as Sumsal
,SUM(salary) over (Partition by gender order by salary) as RunningSalary
from OVER_TEST
